 81    def get_icd9(self, codes: list, unique_eids=True):
 82        """
 83        Returns a DataFrame of all eids with matching ICD9 records
 84
 85        Args:   
 86        - codes (list): List of ICD9 codes for filtering
 87
 88        Returns:
 89            DataFrame: Containing all matching eids for given ICD9 codes
 90        """
 91        distinct = "DISTINCT" if unique_eids else ""
 92        # ICD9: remove . from codes, first 5 characters only
 93        formatted_codes = ", ".join([f"'{code.replace('.', '')[:5]}'" for code in codes])
 94        df = self.spark.sql(f"SELECT {distinct} eid, dnx_hesin_id FROM `{self.database}`.`hesin_diag` WHERE diag_icd9 IN ({formatted_codes})")

117    def get_hesin_oper(self, codes: list, unique_eids=True, earliest_date=True):
118        distinct = "DISTINCT" if unique_eids else ""
119        # OPCS3, OPCS4: remove the . from codes, first 5 chars only

121        df = self.spark.sql(f"SELECT {distinct} eid, opdate FROM `{self.database}`.`hesin_oper` WHERE oper3 IN ({formatted_codes}) OR oper4 in ({formatted_codes})")
122        
123        # get earliest available date for duplicate eids
124        if earliest_date:

144    def get_hesin_dates(self, df: DataFrame, earliest_date=True):
145
146        hesin_df = self.spark.sql(f"SELECT dnx_hesin_id, epistart, epiend, admidate, disdate FROM `{self.database}`.`hesin`")
147
148        df = df.join(hesin_df, on="dnx_hesin_id", how="inner")
149
150        # get diagnosis date: takes the first non-null value since fields are in chronological order
151        df = df.withColumn('diagnosis_date', F.coalesce('epistart', 'epiend', 'admidate', 'disdate'))
152
153        # group by eid and get the earliest diagnosis year for each participant
154        df = df.groupBy('eid').agg(F.min('diagnosis_date').alias('diagnosis_date'))
155
156        return df.select("eid", "diagnosis_date")

174    def get_cancer(self, codes: list, unique_eids=True, earliest_date=True):
175        distinct = "DISTINCT" if unique_eids else ""
176        formatted_codes = ", ".join([f"'{code.replace('.', '')[:5]}'" for code in codes])
177
178        # put fields into SQL query
179        case_statements = [f"WHEN p40006_i{i} IN ({formatted_codes}) THEN p40005_i{i}" for i in range(22)]
180        case_condition = " ".join(case_statements)
181
182        query = f"""
183        SELECT {distinct} eid, 
184               CASE {case_condition} 
185               ELSE NULL 
186               END AS cancer_date
187        FROM `{self.database}`.`participant_0073`
188        WHERE { " OR ".join([f"p40006_i{i} IN ({formatted_codes})" for i in range(22)]) }
189        """
190
191        df = self.spark.sql(query)
192
193        # get earliest available date for duplicate eids
194        if earliest_date:
195            df = df.groupBy("eid").agg(F.min("cancer_date").alias("cancer_date"))
196            
197        return df

201    def get_rap_phenos(self, codes: list):
202        with open('field_lookup.json', 'r') as json_file:
203            lookup_dict = json.load(json_file)
204        lookup = [(code, lookup_dict[code]) for code in codes]
205        del lookup_dict
206        
207        if len(lookup) == 0:
208            print("Error: Fields not found...")
209            return
210        
211        for code, table in lookup:
212            print(f"{code} found in {table}")
213        
214        code, table = lookup[0]
215        df = self.spark.sql(f"SELECT eid, {code} FROM `{self.database}`.`{table}`")    
216
217        for code, table in lookup[1:]:
218            new_df = self.spark.sql(f"SELECT eid, {code} FROM `{self.database}`.`{table}`")
219            df = df.join(new_df, on="eid", how="outer")
220
221        return df

226    def get_database_from_nexus(self):
227        dataset_file = None
228        try:
229            # execute the `dx ls` command
230            result = subprocess.run(
231                ["dx", "ls", "/"],
232                text=True,
233                capture_output=True,
234                check=True
235            )
236
237            files = result.stdout.splitlines()
238            dataset_file = [file[:-8] for file in files if file.endswith('.dataset')][0]
239
240        except subprocess.CalledProcessError as e:
241            print(f"Error listing DNA nexus databases: {e.stderr}\nPlease define database manually using set_database(database_name)")
242            self.find_databases()
243
244        if dataset_file:
245            print(f"Running on database {dataset_file}...")
246            self.database = dataset_file
247        else:
248            print("No .dataset files found in nexus directory... Please define database manually using set_database(database_name)")
249            self.find_databases()

254            databases = [row.namespace for row in self.spark.sql("SHOW DATABASES").collect()]
255            print(f"Found {len(databases)} available databases:")
256            for i, db in enumerate(databases):
257                print(f"{i + 1}. {db}")

260            print(f"Error finding available databases: {e}")
