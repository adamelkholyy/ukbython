 34    def get_icd9(self, codes: list, unique_eids=True):
 35        """
 36        Returns a DataFrame of all eids and the first diagnosis date of matching ICD9 records.
 37
 38        **Args**
 39        - `codes` (`list`): List of ICD9 codes for filtering
 40        - `unique_eids` (`bool`): If True, return only unique eids with the earliest diagnosis date for each
 41
 42        **Returns**
 43        - `DataFrame`: Containing all matching eids and earliest diagnosis date for given ICD9 codes
 44        """
 45        start = time.time()
 46        print(f"Searching {len(codes)} ICD9 codes...")
 47        
 48        # ICD9: remove the . from codes, first 5 characters only
 49        formatted_codes = ", ".join([f"'{code.replace('.', '')[:5]}'" for code in codes])
 50        
 51        df = self.spark.sql(f"SELECT DISTINCT eid, dnx_hesin_id FROM `{self.database}`.`hesin_diag` WHERE diag_icd9 IN ({formatted_codes})")
 52        df = self.get_hesin_dates(df, earliest_date=unique_eids)
 53        print(f"Operation complete: Found {df.count()} matching records in {(time.time() - start):.2f} seconds.")
 54        
 55        return df

 58    def get_icd10(self, codes:list, unique_eids=True):
 59        """
 60        Returns a DataFrame of all eids and the first diagnosis date of matching ICD10 records across HESIN, Death, and Cancer records.
 61
 62        **Args**
 63        - `codes` (`list`): List of ICD10 codes for filtering
 64        - `unique_eids` (`bool`): If True, return only unique eids with the earliest diagnosis date for each
 65
 66        **Returns**
 67        - `DataFrame`: Containing all matching eids and earliest diagnosis date for given ICD10 codes
 68        """
 69        start = time.time()
 70        print(f"Searching {len(codes)} ICD10 codes...")
 71        hesin_df = self.get_icd10_hesin(codes, earliest_date=unique_eids)
 72        print(f"Found {hesin_df.count()} matching hospital records")
 73
 74        death_df = self.get_icd10_death(codes)
 75        print(f"Found {death_df.count()} matching death records")
 76
 77        cancer_df = self.get_icd10_cancer(codes, unique_eids=unique_eids)
 78        print(f"Found {cancer_df.count()} matching cancer records")
 79
 80        # join dfs together
 81        df = hesin_df.join(death_df, on="eid", how="outer")
 82        df = df.join(cancer_df, on="eid", how="outer")
 83
 84        # get earliest date of diagnosis if unique_eids == True
 85        if unique_eids:
 86            df = df.withColumn(
 87                "diagnosis_date",
 88                F.least(
 89                    F.col("diagnosis_date"),
 90                    F.col("cancer_date"),
 91                    F.col("date_of_death")
 92                )
 93            )
 94            df = df.select("eid", "diagnosis_date")
 95            
 96        print(f"Operation complete: Found {df.count()} matching records in {(time.time() - start):.2f} seconds.")
 97        return df

138    def get_icd10_cancer(self, codes: list, unique_eids=True):
139        """
140        Returns a DataFrame of all eids and the first diagnosis date of matching ICD10 diagnoses from cancer records.
141
142        **Args**
143        - `codes` (`list`): List of ICD10 codes for filtering
144        - `unique_eids` (`bool`): If True, return only unique eids with the earliest diagnosis date for each
145
146        **Returns**
147        - `DataFrame`: Containing all matching eids and earliest diagnosis date for given ICD10 codes
148        """
149
150        # ICD10: remove . from codes, first 5 characters only
151        formatted_codes = ", ".join([f"'{code.replace('.', '')[:5]}'" for code in codes])
152
153        # generate all cancer fields query
154        case_statements = [f"WHEN p40006_i{i} IN ({formatted_codes}) THEN p40005_i{i}" for i in range(22)]
155        case_condition = " ".join(case_statements)
156
157        # query all cancer fields and associated dates
158        query = f"""
159        SELECT DISTINCT eid, 
160               CASE {case_condition} 
161               ELSE NULL 
162               END AS cancer_date
163        FROM `{self.database}`.`participant_0073`
164        WHERE {" OR ".join([f"p40006_i{i} IN ({formatted_codes})" for i in range(22)])}
165        """
166
167        df = self.spark.sql(query)
168
169        # get earliest available date for duplicate eids
170        if unique_eids:
171            df = df.groupBy("eid").agg(F.min("cancer_date").alias("cancer_date"))
172            
173        return df

229    def get_hesin_dates(self, df: DataFrame, earliest_date=True):
230        """
231        Returns a DataFrame of diagnosis dates of matching eids from hospital inpatient records.
232
233        **Args**
234        - `df` (`DataFrame`): DataFrame of eids to retreive diagnosis dates for 
235        - `earliest_date` (`bool`): If True, return only the earliest diagnosis date for each eid
236
237        **Returns**
238        - `DataFrame`: Containing all diagnosis dates for given eids
239        """
240        hesin_df = self.spark.sql(f"SELECT dnx_hesin_id, epistart, epiend, admidate, disdate FROM `{self.database}`.`hesin`")
241
242        df = df.join(hesin_df, on="dnx_hesin_id", how="inner")
243
244        # get diagnosis date: takes the first non-null value since fields are in chronological order
245        df = df.withColumn('diagnosis_date', F.coalesce('epistart', 'epiend', 'admidate', 'disdate'))
246
247        # group by eid and get the earliest diagnosis date for each participant
248        if earliest_date:
249            df = df.groupBy('eid').agg(F.min('diagnosis_date').alias('diagnosis_date'))
250
251        return df.select("eid", "diagnosis_date")

275    def get_rap_field(self, codes: list):
276        """
277        Returns a DataFrame of eids and given RAP fields.
278
279        **Args**
280        - `codes` (`list`): List of RAP fields for filtering
281
282        **Returns**
283        - `DataFrame`: Containing all database eids and specified RAP fields
284        """
285        start = time.time()
286        print(f"Searching {len(codes)} RAP fields...")
287
288        # validation for field lookup table
289        if not os.path.isfile('field_lookup.json'):
290            raise FileNotFoundError('field_lookup.json not found... Try running:\n curl -L -o field_lookup.json "https://raw.githubusercontent.com/adamelkholyy/ukbython/main/field_lookup.json"')
291
292        # load the field lookup dictionary for fast lookup of RAP fields
293        with open('field_lookup.json', 'r') as json_file:
294            lookup_dict = json.load(json_file)
295        lookup = [(code, lookup_dict[code]) for code in codes]
296        del lookup_dict
297        
298        # validation for dictionary loading
299        if len(lookup) == 0:
300            raise ValueError("Fields not found...")
301        
302        # print the tables where fields are found for user's reference
303        for code, table in lookup:
304            print(f"{code} found in {table}")
305        
306        # join given fields together with all eids
307        code, table = lookup[0]
308        df = self.spark.sql(f"SELECT eid, {code} FROM `{self.database}`.`{table}`")    
309        for code, table in lookup[1:]:
310            new_df = self.spark.sql(f"SELECT eid, {code} FROM `{self.database}`.`{table}`")
311            df = df.join(new_df, on="eid", how="outer")
312            
313        print(f"Operation complete: Found {df.count()} matching records in {(time.time() - start):.2f} seconds.")
314        return df

318    def get_database_from_nexus(self):
319        """
320        Searches the DNANexus directory for all viable database files. Sets the first one found if available, otherwise prompts the user to set the database manually.
321        """
322        dataset_file = None
323        try:
324            # execute the `dx ls` command
325            result = subprocess.run(
326                ["dx", "ls", "/"],
327                text=True,
328                capture_output=True,
329                check=True
330            )
331            # sets the first .dataset file found as database
332            files = result.stdout.splitlines()
333            dataset_file = [file[:-8] for file in files if file.endswith('.dataset')][0]
334
335        # error: cannot list databases, list databases manually instead
336        except subprocess.CalledProcessError as e:
337            self.find_databases()
338            raise e(f"Error listing DNA nexus databases: {e.stderr}. Please define database manually from any of the available sets above using set_database(database_name)")
339
340        # dataset file found
341        if dataset_file:
342            print(f"Running on database {dataset_file}...")
343            self.database = dataset_file
344        # error: no dataset files found, list databases manually instead
345        else:
346            self.find_databases()
347            raise ValueError(f"No .dataset files found in nexus directory... Please define database manually from any of the available sets above using set_database(database_name)")

350    def find_databases(self):
351        """
352        Lists available databases in the PySpark session.
353        """
354        try:
355            # list databases using PySpark
356            databases = [row.namespace for row in self.spark.sql("SHOW DATABASES").collect()]
357            print(f"Found {len(databases)} available databases:")
358            for i, db in enumerate(databases):
359                print(f"{i + 1}. {db}")
360        # error: no databases found
361        except Exception as e:
362            raise e(f"Error finding available databases: {e.stderr}")
