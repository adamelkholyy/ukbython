 62    def get_database_from_nexus(self):
 63        dataset_file = None
 64        try:
 65            # execute the `dx ls` command
 66            result = subprocess.run(
 67                ["dx", "ls", "/"],
 68                text=True,
 69                capture_output=True,
 70                check=True
 71            )
 72
 73            files = result.stdout.splitlines()
 74            dataset_file = [file[:-8] for file in files if file.endswith('.dataset')][0]
 75
 76        except subprocess.CalledProcessError as e:
 77            print(f"Error listing DNA nexus databases: {e.stderr}\nPlease define database manually using set_database(database_name)")
 78            self.find_databases()
 79
 80        if dataset_file:
 81            print(f"Running on database {dataset_file}...")
 82            self.database = dataset_file
 83        else:
 84            print("No .dataset files found in nexus directory... Please define database manually using set_database(database_name)")
 85            self.find_databases()

113    def get_hesin_dates(self, df: DataFrame, name: str):
114
115        hesin_df = self.spark.sql(f"SELECT dnx_hesin_id, epistart, epiend, admidate, disdate FROM `{self.database}`.`hesin`")
116
117        df = df.join(hesin_df, on="dnx_hesin_id", how="inner")
118
119        # get diagnosis date: takes the first non-null value since fields are in chronological order
120        df = df.withColumn('diagnosis_date', F.coalesce('epistart', 'epiend', 'admidate', 'disdate'))
121
122        # extract the year from the diagnosis_date
123        df = df.withColumn('diagnosis_year', F.year('diagnosis_date'))
124
125        # group by eid and get the earliest diagnosis year for each participant
126        df = df.groupBy('eid').agg(F.min('diagnosis_year').alias(f'{name}_first_diagnosis_date'))
127
128        return df.select("eid", f'{name}_first_diagnosis_date')

190    def get_rap_phenos(self, codes: list):
191        with open('field_lookup.json', 'r') as json_file:
192            lookup_dict = json.load(json_file)
193        lookup = [(code, lookup_dict[code]) for code in codes]
194        del lookup_dict
195        
196        if len(lookup) == 0:
197            print("Error: Fields not found...")
198            return
199        
200        code, table = lookup[0]
201        df = self.spark.sql(f"SELECT eid, {code} FROM `{self.database}`.`{table}`")    
202
203        for code, table in lookup[1:]:
204            new_df = self.spark.sql(f"SELECT eid, {code} FROM `{self.database}`.`{table}`")
205            df = df.join(new_df, on="eid", how="outer")
206
207        return df
